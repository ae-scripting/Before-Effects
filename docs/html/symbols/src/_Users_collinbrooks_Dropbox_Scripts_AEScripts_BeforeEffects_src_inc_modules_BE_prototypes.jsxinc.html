 15 if (!Array.hasOwnProperty("inArray")) {
 16     BE.LOG.debug("Defining Array.inArray()");
 17     /**
 18      * Determines whether or not a given value is within a given array
 19      * @param {Any} needle The needle to search for
 20      * @param {Array} haystack The array (haystack) to search for the needle in.
 21      * @return {Bool} True if the needle is found, false if it isn't
 22      */
 23     Array.prototype.inArray = function (needle, haystack) {
 24         var i;
 25         for (i in haystack)
 26         {
 27             if (haystack[i] === needle)
 28             {
 29                 return true;
 30             }
 31         }
 32         return false;
 33     };
 34 }

 36 if (!Array.hasOwnProperty("filter")) {
 37     /**
 38      * Filters an array based upon a filter function.
 39      * @param {Array} a The array to filter
 40      * @param {Function} f The function to use when filtering the array. If the
 41      * function returns true, the item in the array is included in the final
 42      * filtered list. If it is false, it is ignored. This function should allow
 43      * for one argument only.
 44      * @returns {Array}
 45      */
 46     Array.prototype.filter = function (a, f) {
 47         var newArray, i;
 48         for (i in a)
 49         {
 50             if (f(i))
 51             {
 52                 newArray[newArray.length] = i;
 53             }
 54         }
 55         return newArray;
 56     };
 57 }

 63 if (!ItemCollection.hasOwnProperty("inItemCollection")) {
 64 
 65     /**
 66      *Determines whether or not a given name matches any of the Item names in
 67      *the given ItemCollection. After Effect's ItemCollection is like an array
 68      *but is not. Unfortunately, this means (beside the fact that the inArray
 69      *function is currently not supported by default) there is no default way
 70      *finding whether or not an item exists within a given collection.
 71      *@param {String} n The name to search for within the ItemCollection
 72      *@param {ItemCollection} coll The ItemCollection to search.
 73      *@param {Bool} dupDetect True to make sure the given name is only found
 74      *once. False to ignore whether or not there are multiple Items with the
 75      *given name.
 76      *@returns {Bool|null} If <code>dupDetect</code> is false, and at least one
 77      *item with the given name is found, true is returned. If
 78      *<code>dupDetect</code> is true, and only one item with the given name is
 79      *found, true is returned. False is returned when an Item with the given
 80      *name cannot be found at all. Null is returned when <code>dupDetect</code>
 81      *is set to true and multiple Items are found with the given name.
 82      *
 83      *TODO:
 84      *-Allow for more specific duplicate detection based upon the type of the
 85      *item
 86      */
 87     ItemCollection.prototype.inItemCollection = function (n, coll, dupDetect) {
 88         var i,
 89         index = {};
 90 
 91         if (!dupDetect)
 92         {
 93             dupDetect = false;
 94         }
 95         
 96         for (i = 1; i <= coll.length; i += 1) {
 97 
 98             //Keep track of the items?
 99             if (dupDetect) {
100                 if (index[i])
101                 {
102                     //Item exists already, return null
103                     return null;
104                 } else {
105                     index[i] = coll[i].name;
106                 }
107             }
108 
109             //Does the current Item have the given name?
110             if (coll[i].name === n) {
111                 return true;
112             }
113 
114         }
115         return false;
116     };
117 }
